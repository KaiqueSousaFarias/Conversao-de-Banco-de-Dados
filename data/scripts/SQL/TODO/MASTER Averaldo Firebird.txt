SELECT PRO_COD AS referencia, SEC_DESCR AS descricaoPDV, CASE UNI_NOME WHEN 'UN' THEN 1
	WHEN 'UNID' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'L' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	WHEN 'PCT' THEN 16
	END
	AS idUnidade, PRO_DESCR AS nome, PRO_CODBAR AS gtin,PRO_CUSTO AS  valorCompra, PRO_CUSTO_REAL AS valorCusto, PRO_PRECO AS valorVenda, PRO_SALDO AS estoqueAtual, CASE PRO_ST_ICMS
	WHEN '102' THEN 1
	WHEN '500' THEN 2
	WHEN '300' THEN 3
	END AS idFiscal
, PRO_COD_NCM AS codigoNCM, PRO_CEST AS codigoCEST, GRU_NOME AS descricaoEtiquetas
FROM PRODUTO;
