//PRODUTO

SELECT CODPRODUTO as referencia, BARRAS as gtin, UNIDADE as idunidade, CUSTO as valorcusto, VENDA as valorvenda, ESTOQUEATUAL, DESCRICAO as nome,  CODIGO_CEST as codigoCEST
FROM ACPRODUTO
WHERE DESATIVADO = 'F';

//CLIENTES

SELECT CODESTADO AS UF, TELEFONE AS TELEFONECOMERCIAL, CEP, CNPJ, INSCRICAO AS INSCRICAOESTADUAL, CNPF AS CPF, ENDERECO AS LOGRADOURO, NOME, EMAIL AS EMAILCOMERCIAL, NOME_FANTASIA AS NOMEFANTASIA, NUMERO, CIDADE, COMPLEMENTO, BAIRRO
FROM ACCLIENTE
WHERE INATIVO = 'F';
