PRODUTO

SELECT p.CODIGO AS referencia, p.DESCRICAO AS nome, p.UNIDADE AS idUnidade, p.CUSTO_NOTAFISCAL AS valorCompra, p.PRCUSTO AS valorCusto, p.PRVENDA AS valorVenda, p.SITUACAO_TRIBUTARIA AS idFiscal, p.CODIGO_BARRA AS gtin, p.CODIGONCM, p.CEST AS codigoCEST, e.ESTOQUE AS estoqueAtual
FROM PRODUTOS AS p
LEFT JOIN ESTOQUE AS e 
ON p.CODIGO = e.CODIGO;
