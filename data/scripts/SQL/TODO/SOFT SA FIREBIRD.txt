//PRODUTOS

SELECT p.ID AS referencia, p.NOME, GTIN, NCM AS codigoNCM, CEST AS codigoCEST, NOME_GRUPO AS observacao,
CASE un.SIGLA
WHEN 'UN' THEN 1
WHEN 'KG' THEN 2
WHEN 'LT' THEN 3
END
AS idUnidade, PRC_CUSTO AS valorCusto, PRC_VENDA AS valorVenda,
CASE t.ICMS_SN_ST
WHEN 102 THEN 1
WHEN 500 THEN 2
WHEN 300 THEN 3
END
AS idFiscal, ESTOQUE AS estoqueAtual
FROM EST_PRODUTOS AS p
LEFT JOIN FIS_TRIBUTACOES t
ON p.TRIBUTACAO = t.ID
LEFT JOIN EST_UNIDADES un
ON p.UNIDADE = un.ID ;
