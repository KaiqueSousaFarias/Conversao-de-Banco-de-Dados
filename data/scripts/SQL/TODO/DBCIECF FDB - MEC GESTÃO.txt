// PRODUTO

SELECT p.PRO_CODIGO AS referencia, d.DPR_NOME AS observacao, p.PRO_NOMECOMPLETO AS nome, p.PRO_CODBARRA AS gtin, g.GRU_NOME AS descricaoPDV, s.SUB_NOME AS descricaoEtiquetas,
CASE PRO_EMBALAGEM
	WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	END
	AS idUnidade, PRO_CUSTOENT AS valorCusto, PRO_VENDA AS valorVenda,
	CASE s2.SIT_SITUACAO
	WHEN '102' THEN 1
	WHEN '500' THEN 2
	WHEN '300' THEN 3
	END
	AS idFiscal, PRO_CLASSIFISCAL AS codigoNCM, PRO_CESTNCM AS codigoCEST
FROM PRODUTO p
LEFT JOIN DEPROD d 
ON p.PRO_DEPROD = d.DPR_CODIGO
LEFT JOIN GRUPO g 
ON p.PRO_GRUPO = g.GRU_CODIGO
LEFT JOIN SUBGRUPO s
ON p.PRO_SUBGRUPO = s.SUB_CODIGO
LEFT JOIN SITRIBUTARIA s2
ON p.PRO_SITRIBUTARIA = s2.SIT_ID;

//CLIENTES

FROM CLIENTE;
