//PRODUTO
SELECT CODPROD AS referencia, CODIGO AS gtin, NOMEPROD AS nome, ESTATU AS estoqueAtual,
CASE UNIDADE
	WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
  END
  AS idUnidade, PRECUSTO AS valorCusto, PREVENDA AS valorVenda, s.DESCRICAO AS observacao, PREVENDATAB2
FROM PRODUTO AS p
LEFT JOIN SECAO AS s 
ON s.CODSEC = p.CODSEC
WHERE

//CLIENTE
SELECT NOMECLI, "TELEFONE" 
FROM CLIENTE;
