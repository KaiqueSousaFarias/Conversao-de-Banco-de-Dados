//PRODUTO

SELECT PRODUTO.CODPRODUTO AS referencia, NOME, CODBARRA AS gtin, NCM AS codigoNCM, UNIDADE AS idUnidade, PRECOCUSTO AS valorCusto, PRECOV1 AS valorVenda, CEST as codigoCEST, ESTOQUE.ESTOQUE
FROM PRODUTO
INNER JOIN ESTOQUE 
ON PRODUTO.CODPRODUTO = ESTOQUE.CODPRODUTO
WHERE INATIVO='F';


//CLIENTE

SELECT NOME, FANTASIA AS nomeFantasia, CNPJCPF AS CNPJ, INSCRG AS inscricaoEstadual, ENDERECO AS logradouro, NUMERO, BAIRRO, UF, CODCIDADE AS codigoMunicipio, CEP, TELEFONE AS telefoneComercial, CELULAR AS celularComercial, COMPL AS complemento
FROM CLIENTE WHERE INATIVO = 'F';


//FORNECEDOR

SELECT NOME, FANTASIA AS nomeFantasia, CNPJCPF AS CNPJ, INSCRG AS inscricaoEstadual, ENDERECO AS logradouro, NUMERO, BAIRRO, UF, CODCIDADE AS codigoMunicipio, CEP, TELEFONE1 AS telefoneComercial
FROM FORNECEDOR WHERE INATIVO = 'F';
