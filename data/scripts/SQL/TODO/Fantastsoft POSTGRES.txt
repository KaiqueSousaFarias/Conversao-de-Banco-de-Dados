SELECT codigo as referencia, codigosweda as gtin, descricao as nome,
case unidade
WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
end
as idUnidade, preco as valorVenda, estoque as estoqueAtual, precocusto as valorCusto,
case nfe_csosn
WHEN '102' THEN 1
WHEN '500' THEN 2
WHEN '300' THEN 3
end
as idFiscal, cest as codigoCEST, ncm as codigoNCM
FROM produtos;
