
//PRODUTO

SELECT p.CODIGO AS referencia, p.DESCRICAO AS nome, c.DESCRICAO AS observacao, 
CASE u.SIGLA
	WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	WHEN 'PCT' THEN 16
	END
AS idUnidade, p.CUSTO AS valorCusto, p.CODBARRA AS gtin, p.PRECOVENDA AS valorVenda, p.CODIGO_NCM AS codigoNCM, p.INATIVO AS produtoInativo,
CASE st.CODIGO
	WHEN '102' THEN 1
	WHEN '500' THEN 2
	WHEN '300' THEN 3
	END
AS idFiscal, p.CEST AS codigoCEST
FROM PRODUTO AS p
INNER JOIN CATPRODUTO c 
ON p.ID_CATEGORIA = c.ID
INNER JOIN UNIDADE u 
ON p.ID_UNIDADE = u.ID 
INNER JOIN SIT_TRIBUTARIA st 
ON st.ID = p.ID_SIT_TRIBUTARIA;

FROM CLIENTE;
