PRODUTO

SELECT REPLACE(p.CODIGO, '.', '') AS referencia, p.CODBARRA AS gtin, p.DESCRICAO AS nome, p.UNIDADE AS idUnidade, p.CODIGONCM, ATIVO AS produtoInativo, p.SITTRIBUT, p.CEST AS codigoCEST, e.CUSTOCASA AS valorCusto, e.CUSTOFONTE AS valorCompra, e.PRECOVENDA AS valorVenda
FROM DBCADMER AS p
LEFT JOIN DBESTOQUE AS e
ON e.CODIGO = p.CODIGO
WHERE e.LOJA = 1;

CLIENTES

SELECT NOME, CNPJ, INSCRICAO AS inscricaoEstadual, CPF, LOGRADOURO, NUMERO, COMPLEMENTO, BAIRRO, CEP, CIDADE, UF, TELEFONE AS telefoneComercial, FAX telefoneResidencial, EMAIL AS emailComercial, CONTATO, OBSERVACAO, IDENTIDADE AS registroGeral
FROM DBCADCLI;

FORNECEDORES

SELECT NOME, CNPJ, INSCRICAO AS inscricaoEstadual, CPF, LOGRADOURO, NUMERO, COMPLEMENTO, BAIRRO, CEP, CIDADE, UF, TELEFONE AS telefoneComercial, FAX telefoneResidencial, EMAIL AS emailComercial, CONTATO, OBSERVACAO
FROM DBCADFOR;
