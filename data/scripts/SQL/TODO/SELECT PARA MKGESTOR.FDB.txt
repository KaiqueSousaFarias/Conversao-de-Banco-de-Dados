PESSOAS

SELECT PES_NOME AS NOME, PES_FANTASIA AS nomeFantasia, PES_ENDERECO AS logradouro, PES_COMPLEMENTO AS complemento, PES_BAIRRO AS bairro, PES_POSTAL AS cep, PES_ESTADO AS uf, PES_CPFCNPJ AS CNPJ, PES_IDEINS AS incricaoEstadual, PES_FONE AS telefoneComercial, PES_CELULAR AS celularComercial, PES_EMAIL AS emailComercial, PES_OBSERVA AS observacao, PES_ENDNUMERO AS numero
FROM PESSOAS;

PRODUTOS

SELECT ESTOQUE_PRODUTO.ESP_CODIGO AS referencia, ESTOQUE_PRODUTO.ESP_CODBAR AS gtin ,ESTOQUE_PRODUTO.ESP_DESCRI AS nome, MARCA.MAR_NOME AS observacao,
CASE UNI_CODIGO
WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	END
	AS idUnidade, ESP_NCM AS codigoNCM, ESE_PRECOM AS valorCompra, ESE_PRECUS as valorCusto, ESE_PREVEN AS valorVenda, ESE_PREATA AS valorVendaAtacado,  CST_CODIGO AS idFiscal, ESE_SALDO AS estoqueAtual
FROM ESTOQUE_PRODUTO
INNER JOIN ESTOQUE_EMPRESA ON ESTOQUE_PRODUTO.ESP_IDFCOD = ESTOQUE_EMPRESA.ESE_IDFCOD
INNER JOIN MARCA ON ESTOQUE_PRODUTO.ESP_MARCOD = MARCA.MAR_CODIGO
WHERE ESTOQUE_EMPRESA.ESE_EMPCOD = 1;
