PRODUTOS

SELECT
	p.CODIGO AS referencia,
	p.CODBARRAS AS gtin,
	p.CODIGONCM,
	CASE p.INATIVO
	WHEN 'N' THEN 0 
	WHEN 'S' THEN 1
	END
	AS produtoInativo,
	p.DESCRICAO AS nome,
	CASE p.UNIDADE
	WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	WHEN '' THEN 1
	WHEN 'TB' THEN 1
	WHEN 'FR' THEN 1
	WHEN 'RL' THEN 1
	END
	AS idUnidade,
	p."QUANTIDADE" AS estoqueAtual,
	p.PRECOVENDA AS valorVenda,
	g.NOME AS descricaoPDV,
	s.NOME AS descricaoEtiquetas,
	p.PRECOCUSTO AS valorCusto,
	p.PRECOFABRICA AS valorCompra,
	p.CEST AS codigoCEST,
	a.CSOSNDENTROUF AS idFiscal
FROM PRODUTOS p
LEFT JOIN GRUPOS g
	ON p.CODGRUPO = g.CODIGO
LEFT JOIN SUBGRUPOS s
	ON p.CODSUBGRUPO = s.CODIGO
LEFT JOIN ALIQUOTAS a
	ON p.CODALIQUOTA = a.CODIGO;

CLIENTES

SELECT 
p.CODIGO AS OBSERVACAO,
p.NOME, FANTASIA AS nomeFantasia,
p.ENDERECO AS logradouro,
p.CLINUMEROEND AS numero,
p.COMPLEMENTO,
p.BAIRRO,
c.CIDADE,
p."CEP", 
p.CNPJ_CPF AS cnpj,
p."TELEFONE" AS telefoneComercial, 
p.CELULAR AS celularComercial
FROM CLIENTES p
LEFT JOIN CIDADES c 
ON p.CODCIDADE = c.CODIGO;
