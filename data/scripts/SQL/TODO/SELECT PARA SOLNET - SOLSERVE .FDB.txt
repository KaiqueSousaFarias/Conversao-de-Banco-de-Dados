//PRODUTO
SELECT DISTINCT p.ID_PRODUTO AS OBSERVACAO, p.DESCRICAO AS NOME, p.CODIGO AS REFERENCIA, p.CODIGO_BARRA AS GTIN, u.DESCRICAO AS IDUNIDADE, n.CODIGO AS codigoNCM, p.CUSTO_UNITARIO AS VALORCUSTO, p.PRECO_VENDA_1 AS VALORVENDA, p.CEST AS CODIGOCEST, n.CSOSN AS IDFISCAL
FROM PRODUTOS p
INNER JOIN UNIDADES u
ON p.ID_UNIDADE = u.ID_UNIDADE 
INNER JOIN NCM n
ON p.ID_NCM = n.ID_NCM
WHERE p.INATIVO = 0;
