--CLIENTES:

SELECT CNPJCPF AS CNPJ, IERG AS inscricaoEstadual, RAZAO AS NOME, FANTASIA AS NOMEFANTASIA, ENDERECO AS LOGRADOURO, NUMERO, COMPLEMENTO, BAIRRO, CIDADE, CEP, UF, FONE1 AS TELEFONECOMERCIAL, FONE2 AS TELEFONERESIDENCIAL, CONTATO, EMAIL AS EMAILCOMERCIAL FROM CLIENTE c  ORDER BY PESSOA


-- PRODUTOS

SELECT p.CODIGO AS gtin, p.CODALFA AS referencia, p.DESCRICAO AS nome, d.DESCRICAO AS observacao, 
p.CUSTO as valorcusto, p.PRECO1 AS valorvenda, p.UNIDADE AS idunidade, p.CODNCM AS codigoNCM, 
c.CEST AS codigoCEST, p.BLOQUEADO AS produtoinativo, e.ETOTAL AS estoqueAtual, pco.OPCIONAL as GTIN
from PRODUTO AS p
left join ESTOQUE AS e ON p.CODIGO = e.CODPRODUTO
left join PRODUTO_CODOPCIONAL AS pco ON p.CODIGO = pco.CODPRODUTO
LEFT JOIN DEPARTAMENTO d ON d.CODIGO = p.CODDEPARTAMENTO
LEFT JOIN CEST c ON c.CODIGO = p.CODCEST
ORDER BY p.CODIGO;
