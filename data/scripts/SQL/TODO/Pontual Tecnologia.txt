PRODUTO

SELECT  p.CDPRODUTO AS referencia, p.PRODUTO AS nome, p.CDUNIDADE AS idUnidade, p.CODIGOBARRA AS gtin, p.CLASSFISCAL AS codigoNCM, p.CEST AS codigoCEST, p.SITTRIBUT AS idFiscal, p.GRUPO AS observacao, p.TIPO AS descricaoPDV,
pp.IDPRECO, pp.PRECO AS valorVenda, pp.VLCUSTO AS valorCusto
FROM PRODUTO AS p
LEFT JOIN PRODUTOPRECO AS pp
ON p.CDPRODUTO = pp.CDPRODUTO
WHERE p.INATIVO = 0 AND pp.IDPRECO = 1;

CLIENTE

SELECT CLIENTE AS nome, RAZAO AS nomeFantasia, ENDERECO AS logradouro, SUBBAIRRO AS complemento, BAIRRO, UF, CEP, NUMERO, CDMUNICIPIO AS codigoMunicipio, CGC AS cnpj, INSCEST AS inscricaoEstadual, EMAIL AS emailComercial, NUMERO, OBS AS observacao, TELEFONE 
FROM CLIENTE
WHERE INATIVO = 0;

FORNECEDOR

SELECT FORNECEDOR AS nome, FANTASIA AS nomeFantasia, ENDERECO AS logradouro, COMPLEMENTO, BAIRRO, UF, CEP, NUMERO, CDMUNICIPIO AS codigoMunicipio, CGC AS cnpj, INSCEST AS inscricaoEstadual, EMAIL AS emailComercial, OBS AS observacao, TELEFONE
FROM FORNECEDOR
WHERE INATIVO = 0;
