SELECT p.COD_PRO AS referencia, cb.COD_BAR AS gtin,e.ESTOQUE, VALOR_PRO AS valorVenda,
CASE UNIDADE_SAIDA_PRO
	WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	END
	AS idUnidade, NOME_PRO AS nome, ATIVO_PRO AS produtoInativo, COD_NCM AS codigoNCM, CODIGO_CEST AS codigoCEST,
	CASE CSOSN_CF_EST
		WHEN 102 THEN 1
		WHEN 500 THEN 2
		WHEN 300 THEN 3
	END 
	AS idFiscal
FROM PRODUTO AS p
LEFT JOIN ESTOQUE AS e
ON p.COD_PRO = e.COD_PRO
LEFT JOIN CODIGO_BARRA AS cb
ON p.COD_PRO = cb.COD_PRO;
