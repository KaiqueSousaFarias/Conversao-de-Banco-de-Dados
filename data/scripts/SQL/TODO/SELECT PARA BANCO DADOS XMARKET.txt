PRODUTOS

SELECT
	ESTOQUE.CODIGO AS gtin,
	ESTOQUE.NOME,
	FISICO AS estoqueAtual,
	CUSTO AS valorCusto,
	VENDA AS valorVenda,
	TIPO AS idUnidade,
	NCM AS codigoNCM,
	CEST AS codigoCEST,
	GRUPO.NOME AS OBSERVACAO
FROM ESTOQUE
LEFT JOIN GRUPO
ON ESTOQUE.CDGRUPO = GRUPO.CODIGO;

PESSOAS
SELECT
	NOME,
	CNPJ,
	INSC_EST AS inscricaoEstadual,
	CPF,
	IDENTIDADE AS registroGeral,
	CEP,
	ENDERECO AS logradouro,
	END_NUMERO AS numero,
	END_COMPLEMENTO AS complemento,
	BAIRRO,
	CONTATO,
	EMAIL AS emailComercial,
	RAZAO_SOCIAL AS nomeFantasia
FROM CLIENTES;
