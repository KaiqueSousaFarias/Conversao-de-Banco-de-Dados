PRODUTO

SELECT e.CODIGO referencia, CODIGO_BARRA gtin, e.DESCRICAO nome,
CASE UNIDADE
	WHEN 'UN' THEN 1
	WHEN 'KG' THEN 2
	WHEN 'LT' THEN 3
	WHEN 'CX' THEN 4
	WHEN 'DZ' THEN 5
	WHEN 'FD' THEN 6
	WHEN 'GL' THEN 7
	WHEN 'GR' THEN 8
	WHEN 'BD' THEN 9
	WHEN 'MT' THEN 10
	WHEN 'M2' THEN 11
	WHEN 'M3' THEN 12
	WHEN 'MG' THEN 13
	WHEN 'ML' THEN 14
	WHEN 'MM' THEN 15
	WHEN 'PC' THEN 16
	WHEN '' THEN 1
	ELSE 1
END AS idUnidade, NCM codigoNCM, CST AS idFiscal, PRECO_COMPRA valorCompra, PRECO_VENDA valorVenda, QUANT_ATUAL estoqueAtual, CEST codigoCEST, g.DESCRICAO observacao
FROM ESTOQUE e
LEFT JOIN GRUPO g
ON e.COD_GRUPO = g.CODIGO;
